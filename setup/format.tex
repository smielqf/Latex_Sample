\definecolor{dkgreen}{rgb}{0,0.6,0}     % 定义绿色
\definecolor{mauve}{rgb}{0.58,0,0.82}   % 定义紫色
\lstset
{
    language=C++,                       % 这条命令可以让LaTeX排版时将C++键字突出显示
    breaklines,                         % 这条命令可以让LaTeX自动将长的代码行换行排版
    extendedchars=false,                % 这条命令可以解决代码跨页时，章节标题，页眉等汉字不显示的问题
    columns=flexible,                   % 这条命令可以使得代码复制不出现多余的空格
    showstringspaces=false,             % 这条命令可以使得代码中间的字符串不出现多余的空格
    numbers=left,                       % 这条命令可以让代码行数显示在左边
    stepnumber=2,                       % 这条命令可以让代码行数间隔为2
    numbersep=10pt,                     % 这条命令可以控制代码行数到边框的距离
    backgroundcolor=\color{white},      % 这条命令可以设置背景颜色
    showspaces=false,                   % 这条命令可以控制代码中间出现空格符号
    showtabs=false,                     % 这条命令可以控制代码中间不出现制表符
    keywordstyle=\color{blue},          % 这条命令可以设置代码关键词为蓝色
    commentstyle=\color{dkgreen},       % 这条命令可以控制代码注释为绿色
    stringstyle=\color{mauve},          % 这条命令可以控制代码中间的字符串为紫色
    identifierstyle=\color{orange},     % 这条命令可以控制代码中间的标识符为橙色
    frame=single,                       % 这条命令可以控制代码的边框类型
    rulecolor=\color{black},            % 这条命令可以控制代码的边框颜色
    captionpos=b,                       % 这条命令可以控制代码的标题在下方
    morekeywords={main},                % 这条命令可以设置代码的关键词
    xleftmargin=2em,                    % 这条命令可以控制代码的边框到文本左边界的距离
    xrightmargin=2em,                   % 这条命令可以控制代码的边框到文本右边界的距离
    aboveskip=1em,                      % 这条命令可以控制代码的边框到文本上边界的距离
    belowskip=1em,                      % 这条命令可以控制代码的边框到文本下边界的距离
}
